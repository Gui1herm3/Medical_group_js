var botaoMarcado = false;


function marcarBotao() {
  var botao = document.getElementById("botao");
  if (botao.classList.contains("botao-marcado")) {
    botao.classList.remove("botao-marcado");
    botaoMarcado = false;
  } else {
    botao.classList.add("botao-marcado");
    botaoMarcado = true;
  }
}
